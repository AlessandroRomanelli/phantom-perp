@staticmethod
def _classify_tier(abs_oi_drop_pct: float) -> int:
    """Classify OI drop into tier. Tier 1: [2%,4%), Tier 2: [4%,8%), Tier 3: [8%,inf)."""
    if abs_oi_drop_pct >= 8.0:
        return 3
    elif abs_oi_drop_pct >= 4.0:
        return 2
    else:
