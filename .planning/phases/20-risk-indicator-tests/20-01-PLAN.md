# ── Liquidation Distance Threshold ──────────────────────────────────────


class TestLiquidationDistanceThreshold:
    """Verify minimum liquidation distance rejection and acceptance."""

    def test_rejects_position_too_close(self) -> None:
        """A stop-loss at the liquidation price should fail the guard."""
        # LONG at 2000, liq at 1800, stop at 1800 = ON the liq line → fail
        assert stop_is_before_liquidation(
            Decimal("1800"), Decimal("1800"), PositionSide.LONG,
        ) is False

    def test_accepts_safely_distant_long(self) -> None:
        assert stop_is_before_liquidation(
            Decimal("1900"), Decimal("1600"), PositionSide.LONG,
        ) is True

    def test_rejects_stop_beyond_liq_short(self) -> None:
        assert stop_is_before_liquidation(
            Decimal("2600"), Decimal("2400"), PositionSide.SHORT,
        ) is False

    def test_accepts_safely_distant_short(self) -> None:
        assert stop_is_before_liquidation(
            Decimal("2100"), Decimal("2400"), PositionSide.SHORT,
        ) is True

    def test_flat_direction_raises(self) -> None:
        with pytest.raises(ValueError):
            stop_is_before_liquidation(
                Decimal("1900"), Decimal("1600"), PositionSide.FLAT,
            )


# ── Position Sizer Max Equity Bound ────────────────────────────────────


class TestPositionSizerMaxEquityBound:
    """Verify position sizer never exceeds max_position_pct_equity."""

    def test_route_a_40pct_bound(self) -> None:
        """With Route A limits (40% equity), margin never exceeds 40% of equity."""
        limits = _default_limits(
            max_position_pct_equity=Decimal("40"),
            max_position_notional_usdc=Decimal("200000"),
            max_leverage=Decimal("10"),
        )
        size = compute_position_size(
            entry_price=Decimal("2000"),
            conviction=1.0,
            equity=Decimal("10000"),
            used_margin=Decimal("0"),
            existing_positions=[],
            limits=limits,
        )
        notional = size * Decimal("2000")
        margin = notional / Decimal("10")
        assert margin <= Decimal("4000")  # 40% of 10000

    def test_route_b_25pct_bound(self) -> None:
        """With Route B limits (25% equity), margin never exceeds 25% of equity."""
        limits = _default_limits(
            max_position_pct_equity=Decimal("25"),
            max_position_notional_usdc=Decimal("200000"),
            max_leverage=Decimal("5"),
        )
        size = compute_position_size(
            entry_price=Decimal("2000"),
            conviction=1.0,
            equity=Decimal("10000"),
            used_margin=Decimal("0"),
            existing_positions=[],
            limits=limits,
        )
        notional = size * Decimal("2000")
        margin = notional / Decimal("5")
        assert margin <= Decimal("2500")  # 25% of 10000

    def test_high_conviction_still_bounded(self) -> None:
        """Even at conviction=1.0, sizing respects all constraints."""
        limits = _default_limits(max_position_notional_usdc=Decimal("3000"))
        size = compute_position_size(
            entry_price=Decimal("2000"),
            conviction=1.0,
            equity=Decimal("100000"),
            used_margin=Decimal("0"),
            existing_positions=[],
            limits=limits,
        )
        notional = size * Decimal("2000")
        assert notional <= Decimal("3000")

    def test_margin_utilization_caps(self) -> None:
        """When used_margin is high, remaining budget is limited."""
        limits = _default_limits(
            max_margin_utilization_pct=Decimal("70"),
            max_position_notional_usdc=Decimal("200000"),
        )
        # 70% of 10000 = 7000 max margin; 6000 used → only 1000 available
        size = compute_position_size(
            entry_price=Decimal("2000"),
            conviction=1.0,
            equity=Decimal("10000"),
            used_margin=Decimal("6000"),
            existing_positions=[],
            limits=limits,
        )
        notional = size * Decimal("2000")
        margin_used = notional / limits.max_leverage
        assert margin_used <= Decimal("1000")
