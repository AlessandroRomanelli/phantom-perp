class TestATR:
    def test_known_value(self) -> None:
        """ATR with constant high-low range and no gaps should equal that range."""
        n = 20
        highs = np.full(n, 110.0)
        lows = np.full(n, 100.0)
        closes = np.full(n, 105.0)
        result = atr(highs, lows, closes, period=5)
        # With constant HL range of 10 and no gaps, ATR converges to 10
        assert result[-1] == pytest.approx(10.0, abs=0.1)

    def test_atr_always_positive(self) -> None:
        np.random.seed(42)
        n = 50
        closes = np.cumsum(np.random.randn(n)) + 100
        highs = closes + np.abs(np.random.randn(n))
        lows = closes - np.abs(np.random.randn(n))
        result = atr(highs, lows, closes, period=14)
        valid = result[~np.isnan(result)]
        assert all(v > 0 for v in valid)


class TestBollingerBands:
    def test_constant_input(self) -> None:
        """For constant values, upper == middle == lower (std = 0)."""
        values = np.full(30, 100.0)
        bb = bollinger_bands(values, period=20, num_std=2.0)
        idx = 25  # well after warmup
        assert bb.middle[idx] == pytest.approx(100.0)
        assert bb.upper[idx] == pytest.approx(100.0)
        assert bb.lower[idx] == pytest.approx(100.0)

    def test_upper_above_lower(self) -> None:
        np.random.seed(42)
        values = np.cumsum(np.random.randn(50)) + 100
        bb = bollinger_bands(values, period=20)
        for i in range(19, 50):
            if not np.isnan(bb.upper[i]):
                assert bb.upper[i] >= bb.lower[i]


class TestRealizedVolatility:
    def test_constant_price_zero_vol(self) -> None:
        values = np.full(30, 100.0)
        result = realized_volatility(values, period=10)
        # Constant prices → zero log returns → zero vol
        valid = result[~np.isnan(result)]
        for v in valid:
            assert v == pytest.approx(0.0, abs=1e-10)

    def test_positive_for_varying_prices(self) -> None:
        np.random.seed(42)
        values = np.cumsum(np.random.randn(50)) + 100
        values = np.abs(values)  # ensure positive
        result = realized_volatility(values, period=10)
        valid = result[~np.isnan(result)]
        assert all(v > 0 for v in valid)

class TestOBV:
    def test_monotonic_increase(self) -> None:
        """Monotonically increasing closes → OBV should be monotonically increasing."""
        closes = np.array([10.0, 11.0, 12.0, 13.0, 14.0])
        volumes = np.array([100.0, 100.0, 100.0, 100.0, 100.0])
        result = obv(closes, volumes)
        # Each step adds 100: [0, 100, 200, 300, 400]
        assert result[-1] == pytest.approx(400.0)

    def test_flat_prices_no_change(self) -> None:
        closes = np.array([10.0, 10.0, 10.0])
        volumes = np.array([100.0, 200.0, 300.0])
        result = obv(closes, volumes)
        assert result[-1] == pytest.approx(0.0)


class TestVWAPIndicator:
    def test_equal_prices_returns_price(self) -> None:
        """When all prices are equal, VWAP should equal that price."""
        n = 5
        price = 100.0
        h = np.full(n, price)
        l = np.full(n, price)
        c = np.full(n, price)
        v = np.array([100.0, 200.0, 300.0, 400.0, 500.0])
        result = vwap(h, l, c, v)
        assert result[-1] == pytest.approx(price)

    def test_hand_calculated(self) -> None:
        h = np.array([12.0, 14.0])
        l = np.array([10.0, 12.0])
        c = np.array([11.0, 13.0])
        v = np.array([100.0, 200.0])
        result = vwap(h, l, c, v)
        # TP1 = (12+10+11)/3 = 11.0, TP2 = (14+12+13)/3 = 13.0
        # cumTPV = [11*100, 11*100+13*200] = [1100, 3700]
        # cumV = [100, 300]
        # VWAP = [11.0, 12.333...]
        assert result[0] == pytest.approx(11.0)
        assert result[1] == pytest.approx(3700.0 / 300.0, rel=1e-4)

class TestStochastic:
    def test_at_high_returns_100(self) -> None:
        """When close is at the highest high, %K should be 100."""
        highs = np.array([10.0, 12.0, 14.0, 16.0, 18.0, 20.0])
        lows = np.array([8.0, 10.0, 12.0, 14.0, 16.0, 18.0])
        closes = np.array([9.0, 11.0, 13.0, 15.0, 17.0, 20.0])  # last close = highest high
        result = stochastic(highs, lows, closes, k_period=3, d_period=2)
        # At index 5: window highs [16,18,20]=20, lows [14,16,18]=14, close=20
        # K = (20-14)/(20-14)*100 = 100
        assert result.k[5] == pytest.approx(100.0)

    def test_identical_hlc_returns_50(self) -> None:
        """When high==low==close, %K should be 50 (midpoint)."""
        n = 20
        val = np.full(n, 100.0)
        result = stochastic(val, val, val, k_period=5, d_period=3)
        valid_k = result.k[~np.isnan(result.k)]
        for v in valid_k:
            assert v == pytest.approx(50.0)

class TestFundingRateZScore:
    def test_constant_rates_zero_zscore(self) -> None:
        """Constant rates → std=0 → NaN (no z-score possible)."""
        rates = np.full(200, 0.01)
        result = funding_rate_zscore(rates, lookback=168)
        # std=0 means z-score can't be computed
        assert np.isnan(result[-1])

    def test_spike_produces_high_zscore(self) -> None:
        rates = np.full(200, 0.01)
        rates[-1] = 0.10  # 10x spike
        result = funding_rate_zscore(rates, lookback=168)
        assert result[-1] > 2.0  # well above 2 sigma

class TestIndicatorBoundary:
    """Edge cases: empty, single-element, and all-identical series."""

    def test_sma_empty(self) -> None:
        result = sma(np.array([]), 3)
        assert len(result) == 0

    def test_sma_single_element(self) -> None:
        result = sma(np.array([42.0]), 1)
        assert result[0] == pytest.approx(42.0)

    def test_ema_single_element(self) -> None:
        result = ema(np.array([42.0]), 1)
        assert result[0] == pytest.approx(42.0)

    def test_atr_single_bar(self) -> None:
        result = atr(np.array([110.0]), np.array([100.0]), np.array([105.0]), period=1)
        assert result[0] == pytest.approx(10.0)

    def test_obv_single_bar(self) -> None:
        result = obv(np.array([100.0]), np.array([500.0]))
        assert len(result) == 1
        assert result[0] == pytest.approx(0.0)

    def test_rsi_identical_values(self) -> None:
        """All-identical values → zero gains and losses → RSI = 100 (no losses)."""
        values = np.full(30, 50.0)
        result = rsi(values, 14)
        valid = result[~np.isnan(result)]
        for v in valid:
            assert v == pytest.approx(100.0)

    def test_adx_insufficient_data(self) -> None:
        """Fewer data points than period → all NaN."""
        result = adx(np.array([110.0]), np.array([100.0]), np.array([105.0]), period=14)
        assert all(np.isnan(result))
