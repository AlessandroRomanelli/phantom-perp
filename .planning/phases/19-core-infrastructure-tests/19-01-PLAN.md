**Publisher tests:**

1. `test_publish_success` - publish a dict, assert return value is a non-empty string (stream entry ID).
2. `test_publish_serializes_payload` - publish `{"key": "val"}`, then xrange the stream, verify the stored `data` field equals `orjson.dumps({"key": "val"})`.
3. `test_publish_connection_error` - patch `self._redis.xadd` to raise `redis.exceptions.ConnectionError`, assert it propagates.
4. `test_publisher_close` - call `close()`, then verify `fake_redis.connection_pool` is closed (or mock `aclose` and assert it was called).

**Consumer tests (non-reclaim):**

5. `test_subscribe_creates_group` - call `subscribe(["stream:a", "stream:b"], "grp", "c1")`, then verify both groups exist via `xinfo_groups`.
6. `test_subscribe_ignores_busygroup` - create the group first, then call subscribe again; no error raised.
7. `test_listen_yields_messages` - subscribe, add a message to the stream via `fake_redis.xadd`, call `listen()` and collect the first yielded tuple. Assert it equals `(stream_name, msg_id_str, parsed_dict)`. Break after first message to avoid infinite loop (use `async for` with a counter or `anext()`).
8. `test_ack_calls_xack` - subscribe, add+read a message, call `ack(channel, group, msg_id)`, then verify `xpending` shows 0 pending.
9. `test_consumer_close_without_subscribe` - create consumer, call `close()` without ever calling `subscribe()`. Assert no error (reclaim_task is None, aclose still called).

```python
"""Tests for Channel name registry."""

from libs.common.models.enums import Route
from libs.messaging.channels import Channel


def test_unified_channels() -> None:
    """Unified channels have correct static string values."""
    assert Channel.MARKET_SNAPSHOTS == "stream:market_snapshots"
    assert Channel.FUNDING_UPDATES == "stream:funding_updates"
    assert Channel.SIGNALS == "stream:signals"
    assert Channel.ALERTS == "stream:alerts"
    assert Channel.USER_OVERRIDES == "stream:user_overrides"


def test_route_scoped_channels_a() -> None:
    """Route A channels use ':a' suffix."""
    assert Channel.ranked_ideas(Route.A) == "stream:ranked_ideas:a"
    assert Channel.approved_orders(Route.A) == "stream:approved_orders:a"
    assert Channel.exchange_events(Route.A) == "stream:exchange_events:a"
    assert Channel.portfolio_state(Route.A) == "stream:portfolio_state:a"
    assert Channel.funding_payments(Route.A) == "stream:funding_payments:a"


def test_route_scoped_channels_b() -> None:
    """Route B channels use ':b' suffix."""
    assert Channel.ranked_ideas(Route.B) == "stream:ranked_ideas:b"
    assert Channel.approved_orders(Route.B) == "stream:approved_orders:b"
    assert Channel.exchange_events(Route.B) == "stream:exchange_events:b"
    assert Channel.portfolio_state(Route.B) == "stream:portfolio_state:b"
    assert Channel.funding_payments(Route.B) == "stream:funding_payments:b"


def test_all_channels_count() -> None:
    """all_channels returns 5 unified + 1 confirmed + 5*2 route-scoped = 16."""
    channels = Channel.all_channels()
    assert len(channels) == 16
    assert len(set(channels)) == 16  # all unique


def test_confirmed_orders_not_scoped() -> None:
    """confirmed_orders has no route suffix — it is a unified channel."""
    result = Channel.confirmed_orders()
    assert result == "stream:confirmed_orders"
    assert ":a" not in result
    assert ":b" not in result
```
